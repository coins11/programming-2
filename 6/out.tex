int search_tree(int x, struct node *p) {
	struct node *current;
	int boolean = 0;

	current = p;
	while (current != NULL) {
		if (current->data == x) {
			boolean = 1;
			break;
		}
		else if (current->data < x) {
			current = current->right;
		}
		else {
			current = current->left;
		}
	}

	return boolean;
}
\end{lstlisting}
\end{shadebox}

int sumOfTree(struct node *p) {
	if (p == NULL) {
		return 0;
	}
	int sum = p->data;
	sum += sumOfTree(p->right);
	sum += sumOfTree(p->left);

	return sum;
}
\end{lstlisting}
